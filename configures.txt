--------

Apache 1.3.12
-------------
SSL_BASE=/usr/local \
EAPI_MM=/usr/local/src/apache/mm-1.1.3 \
./configure \
--prefix=/usr/local/apache \
--enable-module=ssl \
--enable-module=most \
--enable-shared=max \
--enable-rule=SHARED_CORE \
--enable-rule=SHARED_CHAIN \
--enable-module=so

make
make certificate
sudo make install
